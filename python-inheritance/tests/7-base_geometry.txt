import unittest
from base_geometry import BaseGeometry

class TestBaseGeometry(unittest.TestCase):
    def test_area(self):
        bg = BaseGeometry()
        with self.assertRaises(Exception) as context:
            bg.area()
        self.assertEqual(str(context.exception), "area() is not implemented")

    def test_integer_validator_valid_input(self):
        bg = BaseGeometry()
        self.assertIsNone(bg.integer_validator("test", 5))

    def test_integer_validator_invalid_input(self):
        bg = BaseGeometry()
        with self.assertRaises(TypeError) as context:
            bg.integer_validator("test", "string")
        self.assertEqual(str(context.exception), "f{name} must be an integer")

        with self.assertRaises(TypeError) as context:
            bg.integer_validator("test", 0)
        self.assertEqual(str(context.exception), "f{name} must be greater than 0")

if __name__ == "__main__":
    unittest.main()
